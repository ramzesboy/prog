#include <stdio.h>
#include <windows.h>

// Функція для обчислення кількості чисел із розрядами
// Використовує цифри 5 та 9, в яких три однакові цифри не стоять поруч
// Припускається, що вхідне число digits дорівнює або менше 30
int countNumbers(int digits) {
    if (digits <= 0) {
        return 0;
    }

    int count = 0;             // Загальна кількість чисел
    int prevCount = 1;         // Кількість чисел з попереднім розрядом
    int prevPrevCount = 1;     // Кількість чисел з попереднім до попереднього розрядом

    // Обчислення кількості чисел із заданими розрядами
    for (int i = 2; i <= digits; i++) {
        count = prevCount + prevPrevCount;       // Кількість чисел на поточному розряді
        prevPrevCount = prevCount;               // Оновлення значення кількості чисел на попередньому до попереднього розряді
        prevCount = count;                       // Оновлення значення кількості чисел на попередньому розряді
    }

    return count * 2;   // Повертаємо кількість чисел, помножену на 2, оскільки перший розряд може бути 5 або 9
}

int main() {

    //переклад на кирилицю
    SetConsoleCP(1251);
    SetConsoleOutputCP(1251);


    int digits;
    printf("Введіть кількість розрядів (р): ");
    scanf("%d", &digits);

    int result = countNumbers(digits);
    printf("Кількість чисел із %d розрядами: %d\n", digits, result);

    return 0;
}
